  this.startupVisualGameObject = function(/**Image*/ image, /**Number*/ x, /**Number*/ y, /**Number*/ z)
    {
        this.startupGameObject(x, y, z);

    }
            
 this.shutdownVisualGameObject = function()
    {
        this.image = null;
        this.shutdownGameObject();
    }

        this.shutdownVisualGameObject();
    }
